Reference implementation:

take_inventory.sh

#!/bin/bash

MD5_CMD=/usr/bin/md5sum

FILE_NAME=/acme_tools/ide/inventory.$(date +%Y%m%d_%H%M%S)


echo "Taking inventory of executables... "
echo " note: "
echo "  1) you may get some complaints about /proc files"
echo "  2) will likely get a complaint about /sys/devices/system/memory/probe"
echo " You can safely ignore these complaints"
find /  -type f -perm /111  -exec $MD5_CMD {} \; | sort > $FILE_NAME
echo "...done with executable inventory"

echo " "
echo -n "Taking inventory of /etc..."
find /etc -type f -exec $MD5_CMD {} \; | sort >> $FILE_NAME
echo "done"
